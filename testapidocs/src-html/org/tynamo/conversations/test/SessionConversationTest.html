014package org.tynamo.conversations.test;
015
016import static org.testng.Assert.assertEquals;
017import static org.testng.Assert.assertTrue;
018
019import org.eclipse.jetty.webapp.WebAppContext;
020import org.testng.annotations.BeforeClass;
021import org.testng.annotations.Test;
022import org.tynamo.test.AbstractContainerTest;
023
024import com.gargoylesoftware.htmlunit.html.DomElement;
025import com.gargoylesoftware.htmlunit.html.HtmlPage;
026
027public class SessionConversationTest extends AbstractContainerTest {
028        @BeforeClass
029        public void configureWebClient() {
030                webClient.getOptions().setThrowExceptionOnFailingStatusCode(false);
031                webClient.getOptions().setThrowExceptionOnScriptError(false);
032        }
033
034        @Override
035        public WebAppContext buildContext() {
036                WebAppContext context = new WebAppContext("src/test/webapp", "/");
037                /*
038                 * Sets the classloading model for the context to avoid an strange "ClassNotFoundException: org.slf4j.Logger"
039                 */
040                context.setParentLoaderPriority(true);
041                return context;
042        }
043
044        @Test
045        public void conversationStartedAndOngoing() throws Exception
046        {
047                // Should we reset the conversation to make sure this succeeds?
048                assertTrue(getSecondsLeft() > 58);
049                Thread.sleep(2000);
050                assertTrue(getSecondsLeft() < 59);
051                HtmlPage page = webClient.getPage(BASEURI + "sessionconversation");
052                page = page.getAnchorByName("endconversation").click();
053                assertEquals(60, getSecondsLeft(page));
054        }
055
056        private int getSecondsLeft() throws Exception {
057                return getSecondsLeft((HtmlPage)webClient.getPage(BASEURI + "sessionconversation") );
058        }
059
060        private int getSecondsLeft(HtmlPage page) throws Exception {
061                DomElement element = page.getElementById("secondsLeft");
062                return Integer.parseInt(element.getTextContent());
063        }
064
065}
