001package org.tynamo.conversations.test;
002
003import static org.testng.Assert.assertFalse;
004
005import org.eclipse.jetty.webapp.WebAppContext;
006import org.testng.annotations.BeforeClass;
007import org.testng.annotations.Test;
008import org.tynamo.test.AbstractContainerTest;
009
010import com.gargoylesoftware.htmlunit.html.HtmlPage;
011
012public class NoConversationTest extends AbstractContainerTest {
013        @BeforeClass
014        public void configureWebClient() {
015                webClient.getOptions().setThrowExceptionOnFailingStatusCode(false);
016                webClient.getOptions().setThrowExceptionOnScriptError(false);
017        }
018        
019        @Override
020        public WebAppContext buildContext() {
021                WebAppContext context = new WebAppContext("src/test/webapp", "/");
022                /*
023                 * Sets the classloading model for the context to avoid an strange "ClassNotFoundException: org.slf4j.Logger"
024                 */
025                context.setParentLoaderPriority(true);
026                return context;
027        }
028        
029        @Test
030        public void conversationNotStarted() throws Exception {
031                HtmlPage page = webClient.getPage(BASEURI + "noconversation");
032                boolean sessionCreated = Boolean.parseBoolean(page.getElementById("sessionCreated").getTextContent());
033                assertFalse(sessionCreated);
034        }
035}
