001package org.tynamo.conversations.test.pages;
002
003import java.util.Random;
004
005import org.apache.tapestry5.ComponentResources;
006import org.apache.tapestry5.annotations.Persist;
007import org.apache.tapestry5.ioc.annotations.Inject;
008import org.tynamo.conversations.services.ConversationManager;
009
010public class ParallelConversation {
011        private final Random random = new Random();
012
013        @Inject
014        private ConversationManager conversationManager;
015
016        @Inject
017        private ComponentResources componentResources;
018
019        @Persist("conversation")
020        private int target;
021
022        private String conversationId;
023
024        public Object onActivate() {
025                if (target == 0) return createConversation();
026                else return null;
027        }
028
029        private Object createConversation() {
030                conversationId = conversationManager.createConversation(componentResources.getPageName(), 60, false);
031                return this;
032        }
033
034        public Object onActivate(String conversationId) {
035                if (!conversationId.equals(conversationManager.getActiveConversation())) return createConversation();
036
037                if (target == 0) target = random.nextInt(10) + 1;
038
039                this.conversationId = conversationId;
040                return null;
041        }
042
043        String onPassivate() {
044                return conversationId;
045        }
046
047        public int getTarget() {
048                return target;
049        }
050}
