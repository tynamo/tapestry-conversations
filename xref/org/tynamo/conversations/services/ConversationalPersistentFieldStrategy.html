15  public class ConversationalPersistentFieldStrategy implements PersistentFieldStrategy {
16  	/**
17  	 * Prefix used to identify keys stored in the session that are being used to store persistent field data.
18  	 */
19  	static final String PREFIX = "state:";
20  
21  	private final ConversationManager conversationManager;
22  
23  	public ConversationalPersistentFieldStrategy(Request request, ConversationManager conversationManager) {
24  		this.prefix = PREFIX;
25  		this.request = request;
26  		conversationManager.setPagePersistentFieldStrategy(this);
27  		this.conversationManager = conversationManager;
28  	}
29  
30  	private final String prefix;
31  
32  	private final Request request;
33  
34  	private String buildPrefix(String pageName) {
35  		// System.out.println("Conversation id is: " + conversationManager.getActiveConversation());
36  		return prefix + pageName + "_conversation_" + conversationManager.getActiveConversation() + ":";
37  	}
38  
39  	public final Collection<PersistentFieldChange> gatherFieldChanges(String pageName) {
40  		Session session = request.getSession(false);
41  
42  		if (session == null) return Collections.emptyList();
43  
44  		List<PersistentFieldChange> result = newList();
45  
46  		// if conversation is not active don't gather any changes
47  		if (conversationManager.getActiveConversation() == null) return result;
48  
49  		String fullPrefix = buildPrefix(pageName);
50  
51  		for (String name : session.getAttributeNames(fullPrefix)) {
52  			Object persistedValue = session.getAttribute(name);
53  
54  			Object applicationValue = persistedValue == null ? null : convertPersistedToApplicationValue(persistedValue);
55  
56  			PersistentFieldChange change = buildChange(name, applicationValue);
57  
58  			result.add(change);
59  
60  			didReadChange(session, name);
61  		}
62  
63  		return result;
64  	}
65  
66  	public void discardChanges(String pageName) {
67  		Session session = request.getSession(false);
68  
69  		if (session == null || conversationManager.getActiveConversation() == null) return;
70  
71  		String fullPrefix = buildPrefix(pageName);
72  
73  		for (String name : session.getAttributeNames(fullPrefix)) {
74  			session.setAttribute(name, null);
75  		}
76  	}
77  
78  	/**
79  	 * Called after each key is read by {@link #gatherFieldChanges(String)}. This implementation does nothing, subclasses
80  	 * may override.
81  	 * 
82  	 * @param session
83  	 *          the session from which a value was just read
84  	 * @param attributeName
85  	 *          the name of the attribute used to read a value
86  	 */
87  	protected void didReadChange(Session session, String attributeName) {
88  	}
89  
90  	private PersistentFieldChange buildChange(String name, Object newValue) {
91  		String[] chunks = name.split(":");
92  
93  		// Will be empty string for the root component
94  		String componentId = chunks[2];
95  		String fieldName = chunks[3];
96  
97  		return new PersistentFieldChangeImpl(componentId, fieldName, newValue);
98  	}
99  	
100 	// Same as org.apache.tapestry5.ioc.internal.util.InternalUtils.isBlank,
101 	// copied here so the same library would work for T5.1 and T5.2
102 	public static boolean isBlank(String input) {
103 		return input == null || input.length() == 0 || input.trim().length() == 0;
104 	}
105 
106 	public final void postChange(String pageName, String componentId, String fieldName, Object newValue) {
107 		assert !isBlank(pageName); 
108 		assert !isBlank(fieldName); 
109 
110 		// If no active conversation, no changes to post
111 		if (conversationManager.getActiveConversation() == null) return;
112 
113 		Object persistedValue = newValue == null ? null : convertApplicationValueToPersisted(newValue);
114 
115 		StringBuilder builder = new StringBuilder(buildPrefix(pageName));
116 
117 		if (componentId != null) builder.append(componentId);
118 
119 		builder.append(':');
120 		builder.append(fieldName);
121 
122 		Session session = request.getSession(persistedValue != null);
123 
124 		// TAPESTRY-2308: The session will be false when newValue is null and the session
125 		// does not already exist.
126 
127 		if (session != null) {
128 			session.setAttribute(builder.toString(), persistedValue);
129 		}
130 	}
131 
132 	/**
133 	 * Hook that allows a value to be converted as it is written to the session. Passed the new value provided by the
134 	 * application, returns the object to be stored in the session. This implementation simply returns the provided value.
135 	 * 
136 	 * @param newValue
137 	 *          non-null value
138 	 * @return persisted value
139 	 * @see #convertPersistedToApplicationValue(Object)
140 	 */
141 	protected Object convertApplicationValueToPersisted(Object newValue) {
142 		return newValue;
143 	}
144 
145 	/**
146 	 * Converts a persisted value stored in the session back into an application value. This implementation returns the
147 	 * persisted value as is.
148 	 * 
149 	 * @param persistedValue
150 	 *          non-null persisted value
151 	 * @return application value
152 	 * @see #convertPersistedToApplicationValue(Object)
153 	 */
154 	protected Object convertPersistedToApplicationValue(Object persistedValue) {
155 		return persistedValue;
156 	}
