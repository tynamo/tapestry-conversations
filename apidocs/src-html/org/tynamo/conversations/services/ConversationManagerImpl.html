001package org.tynamo.conversations.services;
002
003import java.util.ArrayList;
004import java.util.Collections;
005import java.util.Iterator;
006import java.util.List;
007import java.util.Map;
008import java.util.Map.Entry;
009import java.util.concurrent.ConcurrentHashMap;
010
011import javax.servlet.http.HttpServletRequest;
012
013import org.apache.tapestry5.EventContext;
014import org.apache.tapestry5.services.ComponentEventRequestParameters;
015import org.apache.tapestry5.services.Cookies;
016import org.apache.tapestry5.services.PageRenderRequestParameters;
017import org.apache.tapestry5.services.Request;
018import org.tynamo.conversations.ConversationAware;
019
020public class ConversationManagerImpl implements ConversationManager {
021        // protected so you can override toString() in case user application already uses the same keys
022        // for something else
023        protected enum Keys {
024                _conversationId, conversations
025        };
026
027        private final Request request;
028
029        private final Cookies cookies;
030
031        private ConversationalPersistentFieldStrategy pagePersistentFieldStrategy;
032
033        private HttpServletRequest servletRequest;
034
035        private Map<String, List<ConversationAware>> conversationAwareListeners = new ConcurrentHashMap<String, List<ConversationAware>>();
036
037        @SuppressWarnings("rawtypes")
038        public ConversationManagerImpl(Request request, HttpServletRequest servletRequest, Cookies cookies,
039                Map<Class, ConversationAware> listeners) {
040                this.request = request;
041                this.cookies = cookies;
042                this.servletRequest = servletRequest;
043                for (Entry<Class, ConversationAware> entry : listeners.entrySet()) {
044                        String pageName = entry.getKey().getSimpleName();
045                        addConversationListener(pageName, entry.getValue());
046                }
047        }
048
049        @SuppressWarnings("unchecked")
050        protected Map<String, Conversation> getConversations() {
051                Map<String, Conversation> conversations = (Map<String, Conversation>) request.getSession(true).getAttribute(Keys.conversations.toString());
052                if (conversations == null) {
053                        // use ConcurrentHashMap instead of Collections.synchronizedMap() because the former synchronizes on remove
054                        conversations = new ConcurrentHashMap<String, Conversation>();
055                        request.getSession(true).setAttribute(Keys.conversations.toString(), conversations);
056                }
057                return conversations;
058        }
059
060        public boolean activateConversation(Object parameterObject) {
061                if (parameterObject == null) return false;
062                EventContext activationContext = null;
063                String pageName = null;
064                if (parameterObject instanceof PageRenderRequestParameters) {
065                        activationContext = ((PageRenderRequestParameters) parameterObject).getActivationContext();
066                        pageName = ((PageRenderRequestParameters) parameterObject).getLogicalPageName();
067                } else if (parameterObject instanceof ComponentEventRequestParameters) {
068                        activationContext = ((ComponentEventRequestParameters) parameterObject).getPageActivationContext();
069                        pageName = ((ComponentEventRequestParameters) parameterObject).getActivePageName();
070                }
071
072                String conversationId = null;
073
074                // Try reading the conversation id from a cookie first
075                try {
076                        conversationId = cookies.readCookieValue(pageName.toLowerCase() + ConversationManagerImpl.Keys._conversationId);
077                        if (conversationId != null) {
078                                Conversation conversation = getConversations().get(conversationId);
079                                if (conversation == null) conversationId = null;
080                                else if (!conversation.isUsingCookie()) conversationId = null;
081                        }
082                } catch (NumberFormatException e) {
083                        // Ignore
084                }
085                // If cookie-based conversation isn't available, try activation context
086                if (conversationId == null) if (activationContext != null) try {
087                        conversationId = activationContext.get(String.class, activationContext.getCount() - 1);
088                } catch (RuntimeException e) {
089                        // Ignore
090                }
091
092                return activate(endConversationIfIdle(conversationId));
093        }
094
095        private boolean activate(Conversation conversation) {
096                if (conversation == null) return false;
097                request.setAttribute(Keys._conversationId.toString(), conversation.getId());
098                return true;
099        }
100
101        public String createConversation(String pageName, Integer maxIdleSeconds) {
102                return createConversation(pageName, maxIdleSeconds, false);
103        }
104
105        public String createConversation(String pageName, Integer maxIdleSeconds, boolean useCookie) {
106                return createConversation(String.valueOf(System.currentTimeMillis()), pageName, maxIdleSeconds, 0, useCookie);
107        }
108
109        public String createConversation(String pageName, Integer maxIdleSeconds, Integer maxConversationLengthSeconds, boolean useCookie) {
110                return createConversation(String.valueOf(System.currentTimeMillis()), pageName, maxIdleSeconds, maxConversationLengthSeconds, useCookie);
111        }
112
113        public String createConversation(String id, String pageName, Integer maxIdleSeconds, Integer maxConversationLengthSeconds, boolean useCookie) {
114                pageName = pageName == null ? "" : pageName;
115                if (useCookie)
116                        cookies.getBuilder(pageName.toLowerCase() + Keys._conversationId.toString(), String.valueOf(id)).write();
117                Conversation conversation = new Conversation(servletRequest.getSession(true).getId(), id, pageName, maxIdleSeconds, maxConversationLengthSeconds, useCookie);
118                endIdleConversations();
119                getConversations().put(id, conversation);
120                activate(conversation);
121                if (conversationAwareListeners.containsKey(conversation.getPageName())) {
122                        List<ConversationAware> conversationListeners = conversationAwareListeners.get(conversation.getPageName());
123                        for (ConversationAware conversationAware : conversationListeners) conversationAware.onConversationCreated(conversation);
124                }
125                return id;
126        }
127
128        public void endIdleConversations() {
129                Iterator<Conversation> iterator = getConversations().values().iterator();
130                while (iterator.hasNext()) {
131                        Conversation conversation = iterator.next();
132                        if (conversation.isIdle(false)) {
133                                discardConversation(conversation, true);
134                                iterator.remove();
135                        }
136                }
137        }
138
139        protected void discardConversation(Conversation conversation, boolean expired) {
140                if (conversation == null) return;
141                // Notify conversation ending
142                if (conversationAwareListeners.containsKey(conversation.getPageName())) {
143                        List<ConversationAware> conversationListeners = conversationAwareListeners.get(conversation.getPageName());
144                        for (ConversationAware conversationAware : conversationListeners) conversationAware.onConversationEnded(conversation, expired);
145                }
146                //discardConversation
147                if (conversation.isUsingCookie()) cookies.removeCookieValue(String.valueOf(conversation.getId()));
148                if (pagePersistentFieldStrategy != null) pagePersistentFieldStrategy.discardChanges(conversation.getPageName());
149        }
150
151        public boolean exists(String conversationId) {
152                Conversation conversation = getConversations().get(conversationId);
153                if (conversation == null) return false;
154                else return true;
155        }
156
157        protected Conversation endConversationIfIdle(String conversationId) {
158                if (conversationId == null) return null;
159                Conversation conversation = getConversations().get(conversationId);
160                if (conversation == null) return null;
161                boolean resetTimeout = !("false".equals(request.getParameter(Parameters.keepalive.name())));
162                if (conversation.isIdle(resetTimeout)) {
163                        discardConversation(conversation, true);
164                        getConversations().remove(conversation.getId());
165                        conversationId = null;
166                }
167                return conversation;
168        }
169
170        public String getActiveConversation() {
171                String conversationId = (String) request.getAttribute(Keys._conversationId.toString());
172                if (conversationId == null) return null;
173                return exists(conversationId) ? conversationId : null;
174        }
175
176        public int getSecondsBeforeActiveConversationBecomesIdle() {
177                String conversationId = getActiveConversation();
178                if (conversationId == null) return -1;
179                Conversation conversation = getConversations().get(conversationId);
180                if (conversation == null) return -1;
181                return conversation.getSecondsBeforeBecomesIdle();
182        }
183
184        public void setPagePersistentFieldStrategy(ConversationalPersistentFieldStrategy pagePersistentFieldStrategy) {
185                this.pagePersistentFieldStrategy = pagePersistentFieldStrategy;
186        }
187
188        public boolean isActiveConversation(String conversationId) {
189                if (conversationId == null) return false;
190                return conversationId.equals(getActiveConversation());
191        }
192
193        public String endConversation(String conversationId) {
194                Conversation conversation = getConversations().get(conversationId);
195                if (conversation == null) return null;
196                discardConversation(conversation, false);
197                getConversations().remove(conversation.getId());
198                return null;
199        }
200
201        public void addConversationListener(String pageName, ConversationAware conversationAware) {
202                List<ConversationAware> conversationListenersForPage = conversationAwareListeners.get(pageName);
203                if (conversationListenersForPage == null) {
204                        conversationListenersForPage = Collections.synchronizedList(new ArrayList<ConversationAware>() );
205                        conversationAwareListeners.put(pageName, conversationListenersForPage);
206                }
207                conversationListenersForPage.add(conversationAware);
208        }
209
210        public void removeConversationListener(String pageName, ConversationAware conversationAware) {
211                List<ConversationAware> conversationListenersForPage = conversationAwareListeners.get(pageName);
212                if (conversationListenersForPage == null) return;
213                conversationListenersForPage.remove(conversationAware);
214        }
215}
