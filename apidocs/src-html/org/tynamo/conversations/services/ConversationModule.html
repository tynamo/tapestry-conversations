001package org.tynamo.conversations.services;
002
003import java.lang.reflect.Method;
004
005import org.apache.tapestry5.SymbolConstants;
006import org.apache.tapestry5.ioc.Configuration;
007import org.apache.tapestry5.ioc.MappedConfiguration;
008import org.apache.tapestry5.ioc.MethodAdviceReceiver;
009import org.apache.tapestry5.ioc.ServiceBinder;
010import org.apache.tapestry5.ioc.annotations.Contribute;
011import org.apache.tapestry5.ioc.annotations.Match;
012import org.apache.tapestry5.ioc.annotations.Order;
013import org.apache.tapestry5.ioc.services.ApplicationDefaults;
014import org.apache.tapestry5.ioc.services.SymbolProvider;
015import org.apache.tapestry5.plastic.MethodAdvice;
016import org.apache.tapestry5.plastic.MethodInvocation;
017import org.apache.tapestry5.services.LibraryMapping;
018import org.apache.tapestry5.services.PersistentFieldStrategy;
019import org.apache.tapestry5.services.Request;
020import org.apache.tapestry5.services.RequestGlobals;
021
022public class ConversationModule {
023        public static void bind(ServiceBinder binder) {
024                binder.bind(ConversationManager.class);
025        }
026
027        @Contribute(SymbolProvider.class)
028        @ApplicationDefaults
029        public static void setApplicationDefaults(MappedConfiguration<String, Object> configuration) {
030                configuration.add(SymbolConstants.JAVASCRIPT_INFRASTRUCTURE_PROVIDER, "jquery");
031        }
032
033        public static void contributeComponentClassResolver(Configuration<LibraryMapping> configuration) {
034                configuration.add(new LibraryMapping("conversation", "org.tynamo.conversations"));
035        }
036
037        public static void contributePersistentFieldManager(MappedConfiguration<String, PersistentFieldStrategy> configuration, RequestGlobals requestGlobals,
038                        Request request, ConversationManager conversationManager) {
039                configuration.add("conversation", new ConversationalPersistentFieldStrategy(request, conversationManager));
040        }
041
042        // public static <T> T decorateComponentRequestHandler(Class<T> serviceInterface, T delegate, RequestHandlerDecorator decorator) {
043        // return decorator.build(serviceInterface, delegate);
044        // }
045
046        @Match("ComponentRequestHandler")
047        @Order("before:*")
048        public static void adviseComponentRequestHandler(final MethodAdviceReceiver receiver,
049                final ConversationManager conversationManager) {
050                MethodAdvice advice = new MethodAdvice() {
051                        public void advise(MethodInvocation invocation) {
052                                conversationManager.activateConversation(invocation.getParameter(0));
053                                invocation.proceed();
054                        }
055                };
056
057                for (Method method : receiver.getInterface().getMethods())
058                        receiver.adviseMethod(method, advice);
059
060        }
061
062}
