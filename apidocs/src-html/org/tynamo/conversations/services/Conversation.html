001package org.tynamo.conversations.services;
002
003import java.io.Serializable;
004
005public final class Conversation implements Serializable {
006        private static final long serialVersionUID = 333320784727498530L;
007
008        private final String pageName;
009
010        private final String sessionId;
011
012        private final String id;
013
014        private final boolean usingCookie;
015
016        private final long maxIdleSeconds;
017
018        private long lastTouched;
019
020        private long requiredEndTimestamp;
021
022        public boolean isUsingCookie() {
023                return usingCookie;
024        }
025
026        public Conversation(String sessionId, String id, String pageName, Integer maxIdleSeconds,
027                Integer maxConversationLengthSeconds, boolean usingCookie) {
028                if (maxIdleSeconds == null) maxIdleSeconds = 0;
029                if (maxConversationLengthSeconds == null) maxConversationLengthSeconds = 0;
030                this.sessionId = sessionId;
031                this.id = id;
032                this.pageName = pageName;
033                this.usingCookie = usingCookie;
034                this.maxIdleSeconds = maxIdleSeconds;
035                this.requiredEndTimestamp = maxConversationLengthSeconds == 0 ? 0L : System.currentTimeMillis() + maxConversationLengthSeconds * 1000L;
036                touch();
037        }
038
039        public String getId() {
040                return id;
041        }
042
043        public String getSessionId() {
044                return sessionId;
045        }
046
047        public String getPageName() {
048                return pageName;
049        }
050
051        public void touch() {
052                lastTouched = System.currentTimeMillis();
053        }
054
055        public Integer getSecondsBeforeBecomesIdle() {
056                if (maxIdleSeconds <= 0) return null;
057                int secondsLeft = (int) ((maxIdleSeconds - (System.currentTimeMillis() - lastTouched) / 1000L));
058                if (requiredEndTimestamp > 0) secondsLeft = Math.min(secondsLeft, (int)((requiredEndTimestamp - System.currentTimeMillis())/1000L));
059                return secondsLeft < 0 ? 0 : secondsLeft;
060        }
061
062        /**
063         * True if conversation has been idle for too long or past its maxConversationLength, otherwise resets the idletime if
064         * resetIdle is true
065         **/
066        public boolean isIdle(boolean resetIdle) {
067                if (requiredEndTimestamp > 0) {
068                        if (System.currentTimeMillis() > requiredEndTimestamp) return true;
069                }
070                if (maxIdleSeconds < 1) {
071                        if (resetIdle) touch();
072                        return false;
073                }
074                if ((System.currentTimeMillis() - lastTouched) / 1000L > maxIdleSeconds) return true;
075                if (resetIdle) touch();
076                return false;
077        }
078
079}
