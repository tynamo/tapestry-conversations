001package org.tynamo.conversations.components;
002
003import org.apache.tapestry5.ComponentResources;
004import org.apache.tapestry5.Link;
005import org.apache.tapestry5.annotations.AfterRender;
006import org.apache.tapestry5.annotations.Parameter;
007import org.apache.tapestry5.ioc.annotations.Inject;
008import org.apache.tapestry5.json.JSONLiteral;
009import org.apache.tapestry5.json.JSONObject;
010import org.apache.tapestry5.services.Request;
011import org.apache.tapestry5.services.javascript.JavaScriptSupport;
012import org.tynamo.conversations.ConversationModeratorAware;
013import org.tynamo.conversations.services.ConversationManager;
014
015public class ConversationModerator {
016        private static final String eventName = "checkidle";
017
018        @Inject
019        private ComponentResources componentResources;
020
021        @Inject
022        private JavaScriptSupport javaScriptSupport;
023
024        @Parameter("15")
025        private int idleCheck;
026
027        @Parameter("30")
028        private int warnBefore;
029
030        @Parameter(defaultPrefix = "literal")
031        private String warnBeforeHandler;
032
033        @Parameter(defaultPrefix = "literal")
034        private String endedHandler;
035
036        @Parameter("false")
037        private boolean keepAlive;
038
039        JSONObject onCheckidle() {
040                // FIXME check if keepalive is set
041                int nextCheckInSeconds = -1;
042                JSONObject object = new JSONObject();
043                String conversationId = conversationManager.getActiveConversation();
044                // Conversation still exists
045                if (conversationId != null) {
046                        nextCheckInSeconds = conversationManager.getSecondsBeforeActiveConversationBecomesIdle();
047                        // Shouldn't be negative.
048                        if (nextCheckInSeconds < 0) return null;
049                        if (componentResources.getContainer() instanceof ConversationModeratorAware) ((ConversationModeratorAware) componentResources.getContainer()).onConversationIdleCheck();
050                        // If keepalive is true, subtract 1 so conversation will be refreshed before end,
051                        if ("true".equals(request.getParameter(ConversationManager.Parameters.keepalive.name()))) nextCheckInSeconds--;
052                        else {
053                        }
054                        // Negative if warn is disabled
055                        int warnInSeconds = Integer.valueOf(request.getParameter("warn"));
056                        // add 1 , no keepalive
057                        if (warnInSeconds < 0) nextCheckInSeconds++;
058                        else {
059                                warnInSeconds = nextCheckInSeconds - warnInSeconds;
060                                // Change next check time for warn time or warn
061                                if (warnInSeconds > 0) nextCheckInSeconds = warnInSeconds;
062                                // limit how many times you trigger the warn
063                                else if (warnInSeconds > -nextCheckInSeconds) object.put("warn", nextCheckInSeconds);
064
065                        }
066                }
067
068                object.put("nextCheck", nextCheckInSeconds);
069                return object;
070        }
071
072        JSONObject onRefresh() {
073                return null;
074        }
075
076        JSONObject onEnd() {
077                if (componentResources.getContainer() instanceof ConversationModeratorAware) ((ConversationModeratorAware) componentResources.getContainer()).onConversationEnded();
078                conversationManager.endConversation(conversationManager.getActiveConversation());
079                return new JSONObject();
080        }
081
082        @Inject
083        private Request request;
084
085        @Inject
086        private ConversationManager conversationManager;
087
088        @AfterRender
089        public void afterRender() {
090                Link link = componentResources.createEventLink(eventName);
091                String baseURI = link.toAbsoluteURI();
092                int index = baseURI.indexOf(":" + eventName);
093                String defaultURIparameters = baseURI.substring(index + eventName.length() + 1);
094                defaultURIparameters += "".equals(defaultURIparameters) ? "?" : "&";
095                defaultURIparameters += ConversationManager.Parameters.keepalive.name() + "=";
096                baseURI = baseURI.substring(0, index + 1);
097
098                Object warnBeforeHandlerParam = warnBeforeHandler == null ? new JSONLiteral(null) : warnBeforeHandler;
099                Object endedHandlerParam = endedHandler == null ? new JSONLiteral(null) : endedHandler;
100                
101                // System.out.println("Active conversation is " +  conversationManager.getActiveConversation());
102                javaScriptSupport.require("conversation/ConversationModerator") //
103                        .invoke("startConversation") //
104                        .with(baseURI, defaultURIparameters, keepAlive, "", idleCheck, warnBefore, warnBeforeHandlerParam, endedHandlerParam);
105        }
106
107}
