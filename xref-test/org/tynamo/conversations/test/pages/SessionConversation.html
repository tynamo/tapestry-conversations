1   package org.tynamo.conversations.test.pages;
2   
3   import org.apache.tapestry5.ComponentResources;
4   import org.apache.tapestry5.annotations.Meta;
5   import org.apache.tapestry5.annotations.Persist;
6   import org.apache.tapestry5.annotations.Property;
7   import org.apache.tapestry5.ioc.annotations.Inject;
8   import org.tynamo.conversations.services.ConversationManager;
9   
10  @Meta("tapestry.persistence-strategy=conversation")
11  public class SessionConversation {
12  	@Persist("session")
13  	private String conversationId;
14  
15  	@Inject
16  	private ConversationManager conversationManager;
17  
18  	@Inject
19  	private ComponentResources componentResources;
20  
21  	@Persist
22  	@Property
23  	private String comment;
24  
25  	@Property
26  	private Integer secondsLeft;
27  
28  	public void setupRender() {
29  		// Using onActivate() for setting up the conversation is problematic in this case
30  		// This example demonstrates why managing the existing conversation is tricky with T5.1
31  		// Self-referential links in T5.2 will managing this easier. Consider the case where the conversation
32  		// has expired or does not exist, but user submits the form - a new conversation should *not* be created
33  		// so it's easier to just use setupRender().
34  		if (comment == null || !conversationManager.isActiveConversation(conversationId)) {
35  			conversationId = conversationManager.createConversation(componentResources.getPageName(), 60, 60, true);
36  			comment = "";
37  		}
38  		secondsLeft = conversationManager.getSecondsBeforeActiveConversationBecomesIdle();
39  	}
40  
41  	public void onAction() {
42  		conversationManager.endConversation(conversationId);
43  	}
44  
45  }
